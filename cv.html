def faktoriyel(n):
    """n! faktöriyeli hesaplar."""
    if n == 0 or n == 1:    # Temel durum: 0! ve 1! = 1
        return 1
    else:
        return n * faktoriyel(n - 1)  # Kendini çağırarak hesaplar

# Recursive fonksiyon: Fibonacci dizisi hesaplar
def fibonacci(n):
    """Fibonacci dizisinin n'inci elemanını döndürür."""
    if n == 0:
        return 0
    elif n == 1:
        return 1
    else:
        # Önceki iki sayının toplamı
        return fibonacci(n - 1) + fibonacci(n - 2)

    printf("ptr'nin değeri (adres): %p\n", ptr);    // ptr'nin tuttuğu adres (sayi'nin adresi)

    // Dinamik bellek yönetimi: runtime'da (program çalışırken) bellek ayırma

    int *dizi;
    int n;

    printf("Kaç elemanlı dizi istiyorsun? ");
    scanf("%d", &n);

    // malloc ile bellekte n tane int için alan ayırıyoruz
    dizi = (int*) malloc(n * sizeof(int));
    if (dizi == NULL) {  // malloc başarısızsa NULL döner
        printf("Bellek ayırma başarısız!\n");
        return 1;
    }

    // Dizinin elemanlarını kullanıcıdan alalım
    for (int i = 0; i < n; i++) {
        printf("dizi[%d] = ", i);
        scanf("%d", &dizi[i]);
    }

    // Diziyi ekrana yazdıralım
    printf("Girdiğiniz dizi: ");
    for (int i = 0; i < n; i++) {
        printf("%d ", dizi[i]);
    }
    printf("\n");

    // Bellek işimiz bittiğinde free ile ayırdığımız alanı geri veriyoruz
    free(dizi);

    return 0;

// ------ BAĞLI LİSTE ------
struct Node {
    int veri;
    struct Node* sonraki;
};

// Yeni düğüm oluşturur
struct Node* yeniDugum(int veri) {
    struct Node* n = (struct Node*)malloc(sizeof(struct Node));
    n->veri = veri;
    n->sonraki = NULL;
    return n;
}

// Listeyi yazdırır
void listeYazdir(struct Node* bas) {
    while (bas != NULL) {
        printf("%d -> ", bas->veri);
        bas = bas->sonraki;
    }
    printf("NULL\n");
}

// Bağlı listeye sona eleman ekler
void sonaEkle(struct Node** bas, int veri) {
    struct Node* yeni = yeniDugum(veri);
    if (*bas == NULL) {
        *bas = yeni;
        return;
    }
    struct Node* temp = *bas;
    while (temp->sonraki != NULL)
        temp = temp->sonraki;
    temp->sonraki = yeni;
}

// Bağlı listeden ilk elemanı çıkarır
void bastanCikar(struct Node** bas) {
    if (*bas == NULL) return;
    struct Node* temp = *bas;
    *bas = (*bas)->sonraki;
    free(temp);
}

// Bağlı listeden son elemanı çıkarır
void sondanCikar(struct Node** bas) {
    if (*bas == NULL) return;
    if ((*bas)->sonraki == NULL) {
        free(*bas);
        *bas = NULL;
        return;
    }
    struct Node* temp = *bas;
    while (temp->sonraki->sonraki != NULL)
        temp = temp->sonraki;
    free(temp->sonraki);
    temp->sonraki = NULL;
}

// ------ YIĞIN (STACK) ------

#define MAX 100
int stack[MAX];
int top = -1;

// Push: yığına eleman ekler
void push(int x) {
    if (top == MAX - 1) {
        printf("Yığın dolu!\n");
        return;
    }
    stack[++top] = x;
}

// Pop: yığından eleman çıkarır
int pop() {
    if (top == -1) {
        printf("Yığın boş!\n");
        return -1;
    }
    return stack[top--];
}

// ------ KUYRUK (QUEUE) ------

int queue[MAX];
int front = 0, rear = -1, size = 0;

// Kuyruğa eleman ekler (enqueue)
void enqueue(int x) {
    if (size == MAX) {
        printf("Kuyruk dolu!\n");
        return;
    }
    rear = (rear + 1) % MAX;
    queue[rear] = x;
    size++;
}

// Kuyruktan eleman çıkarır (dequeue)
int dequeue() {
    if (size == 0) {
        printf("Kuyruk boş!\n");
        return -1;
    }
    int val = queue[front];
    front = (front + 1) % MAX;
    size--;
    return val;
}

// Linear Search: doğrusal arama yapar
int linearSearch(int arr[], int n, int x) {
    for (int i = 0; i < n; i++)
        if (arr[i] == x)
            return i; // Bulunan indeks
    return -1; // Bulunamadı
}

// Binary Search: sıralı dizide ikili arama yapar (recursive)
int binarySearch(int arr[], int sol, int sag, int x) {
    if (sag >= sol) {
        int mid = sol + (sag - sol) / 2;
        if (arr[mid] == x) return mid;
        if (arr[mid] > x) return binarySearch(arr, sol, mid - 1, x);
        return binarySearch(arr, mid + 1, sag, x);
    }
    return -1; // Bulunamadı
}

// Bubble Sort: basit kabarcık sıralama
void bubbleSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++)
        for (int j = 0; j < n - i - 1; j++)
            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
            }
}

// Selection Sort: seçmeli sıralama
void selectionSort(int arr[], int n) {
    int min_idx;
    for (int i = 0; i < n - 1; i++) {
        min_idx = i;
        for (int j = i + 1; j < n; j++)
            if (arr[j] < arr[min_idx])
                min_idx = j;
        int tmp = arr[min_idx];
        arr[min_idx] = arr[i];
        arr[i] = tmp;
    }
}

// Insertion Sort: eklemeli sıralama
void insertionSort(int arr[], int n) {
    for (int i = 1; i < n; i++) {
        int key = arr[i];
        int j = i - 1;
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

    struct Node* bas = NULL;
    sonaEkle(&bas, 10);
    sonaEkle(&bas, 20);
    sonaEkle(&bas, 30);
    printf("Bağlı Liste: ");
    listeYazdir(bas);

    printf("Bağlı listeden baştan çıkarma...\n");
    bastanCikar(&bas);
    listeYazdir(bas);

    printf("Bağlı listeden sondan çıkarma...\n");
    sondanCikar(&bas);
    listeYazdir(bas);

    // Yığın Örnek
    printf("\nYığın (Stack) işlemleri:\n");
    push(100);
    push(200);
    push(300);
    printf("Pop: %d\n", pop());
    printf("Pop: %d\n", pop());

    // Kuyruk Örnek
    printf("\nKuyruk (Queue) işlemleri:\n");
    enqueue(10);
    enqueue(20);
    enqueue(30);
    printf("Dequeue: %d\n", dequeue());
    printf("Dequeue: %d\n", dequeue());

    // Arama Örnekleri
    int arr[] = {3, 6, 8, 10, 15, 20};
    int n = sizeof(arr) / sizeof(arr[0]);
    int aranan = 10;

    printf("\nLinear Search: Eleman %d dizide ", aranan);
    int idx = linearSearch(arr, n, aranan);
    if (idx != -1)
        printf("bulundu, index: %d\n", idx);
    else
        printf("bulunamadı.\n");

    printf("Binary Search: Eleman %d dizide ", aranan);
    idx = binarySearch(arr, 0, n - 1, aranan);
    if (idx != -1)
        printf("bulundu, index: %d\n", idx);
    else
        printf("bulunamadı.\n");

    // Sıralama Örnekleri
    int dizi1[] = {64, 34, 25, 12, 22, 11, 90};
    int dizi2[] = {64, 34, 25, 12, 22, 11, 90};
    int dizi3[] = {64, 34, 25, 12, 22, 11, 90};
    int size = sizeof(dizi1) / sizeof(dizi1[0]);

    bubbleSort(dizi1, size);
    printf("\nBubble Sort sonucu: ");
    diziYazdir(dizi1, size);

    selectionSort(dizi2, size);
    printf("Selection Sort sonucu: ");
    diziYazdir(dizi2, size);

    insertionSort(dizi3, size);
    printf("Insertion Sort sonucu: ");
    diziYazdir(dizi3, size);

    return 0;
}

# Örnek veri oluştur, öğrenciler ve sınav sonuçları
veri <- data.frame(
  isim = c("Ali", "Ayşe", "Mehmet", "Zeynep", "Can", "Elif"),
  yas = c(20, 22, 19, 21, 23, 22),
  cinsiyet = factor(c("E", "K", "E", "K", "E", "K")),
  not1 = c(85, 90, 78, 92, 67, 88),
  not2 = c(80, 87, 82, 95, 70, 85),
  tarih = as.Date(c("2025-01-15", "2025-02-10", "2025-01-18", "2025-02-12", "2025-01-20", "2025-02-14"))
)

# dplyr ile yaşa göre filtreleme (yas > 20) yap, sadece isim ve not1 seç, ve not1'e göre azalan sırala
sonuc <- veri %>%
  filter(yas > 20) %>%                                 # Yaşı 20'den büyük olanları filtreler
  select(isim, not1) %>%                               # Sadece isim ve not1 sütunlarını seçer
  arrange(desc(not1))                                  # not1'e göre azalan sıralar

print(sonuc)  # sonucu yazdır

# tidyr ile veri şekillendirme: not1 ve not2'yi uzun formata getir
veri_uzun <- veri %>%
  pivot_longer(cols = starts_with("not"), names_to = "sinav", values_to = "puan")  # not1, not2 sütunlarını sinav ve puan olarak dönüştürür

print(veri_uzun)

# ggplot2 ile notların histogramını oluştur
ggplot(veri_uzun, aes(x = puan, fill = sinav)) +
  geom_histogram(binwidth = 5, alpha = 0.7, position = "identity") +  # Puanları 5'erlik aralıklara ayırır ve sinav türüne göre renklendirir
  labs(title = "Sınav Puanlarının Dağılımı", x = "Puan", y = "Frekans") +
  theme_minimal()

# stringr ile isimlerdeki ilk harfi büyük yap
veri <- veri %>%
  mutate(isim = str_to_title(isim))  # isim sütunundaki isimlerin ilk harfi büyük olur

# lubridate ile tarih sütunundan ay bilgisini çıkar
veri <- veri %>%
  mutate(ay = month(tarih, label = TRUE))  # tarih sütunundan ay bilgisini çıkarır ve etiket olarak ekler

print(veri)

# forcats ile cinsiyet faktör seviyelerini yeniden sırala
veri$cinsiyet <- fct_relevel(veri$cinsiyet, "K", "E")  # cinsiyet faktörünü 'K' önce olacak şekilde sıralar

# readr ile CSV dosyasına veri yaz
write_csv(veri, "ogrenciler.csv")  # veri çerçevesini 'ogrenciler.csv' olarak kaydeder

# caret ile basit sınıflandırma modeli için veri hazırlığı
set.seed(123)
train_idx <- createDataPartition(veri$cinsiyet, p = 0.7, list = FALSE)  # Veri %70 eğitim için böl

train <- veri[train_idx, ]  # Eğitim seti
test <- veri[-train_idx, ]  # Test seti

# Random Forest modeli ile cinsiyet tahmini
rf_model <- randomForest(cinsiyet ~ yas + not1 + not2, data = train, ntree = 100)  # Yas, not1, not2 ile cinsiyeti tahmin etmeye çalışır

# Model tahmini

# Başarı ölçütü

print(confusion)

# e1071 paketi ile SVM modeli kur

svm_conf <- table(test$cinsiyet, svm_pred)

print(svm_conf)
